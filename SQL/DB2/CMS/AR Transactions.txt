SELECT 
	T.*, AMT*RATE AMT_USD
FROM 
	RLARP.FFSBGLR1 T
	INNER JOIN LGDAT.MAST ON
		AZCOMP||DIGITS(AZGL#1)||DIGITS(AZGL#2) = ACCT
	LEFT OUTER JOIN RLARP.FFCRET X ON	
		T.PERD = X.PERD AND
		X.FCUR = AZFUT2 AND
		X.TCUR = 'US' AND
		X.RTYP = 'ME'
WHERE
	AZGROP = '1102010' AND 
	T.PERD = '1705'
