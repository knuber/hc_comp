SELECT
	D.COMP, ACC,AZTITL, '' INACTIVE, FSPR PERIOD, 
	OPEN_LOCAL, NET_LOCAL, END_LOCAL, 
	FGRP,
	USD_NET, USD_END,
	OPEN_LOCAL*X.RATE OPEN_PPFX,
	USD_OPEN OPEN_CPFX
FROM
	RLARP.VW_FFTBLCS D
	LEFT OUTER JOIN RLARP.VW_PRFSPR P ON
		P.CURRP = D.FSPR AND
		P.COMP = D.COMP
	LEFT OUTER JOIN RLARP.FFCRET X ON
		FCUR = D.CURR AND
		TCUR = 'US' AND
		RTYP = 'ME' AND
		PERD = P.PRIORP
WHERE
	FGRP LIKE '1102010%' AND
	FSPR >= '1601' AND D.COMP > '19' AND SUBSTR(ACC,11,2) <> '00'
order by fspr asc
